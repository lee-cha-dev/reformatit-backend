# Build & Test Docker Image Locally
docker build -t reformatit-backend .
docker run -p 8080:8080 reformatit-backend


# Push the Docker image to Google Container Registry:
docker tag reformatit-backend gcr.io/reformatit/reformatit-backend
docker push gcr.io/reformatit/reformatit-backend


# Deploy the FastAPI backend to Cloud Run:
gcloud run deploy reformatit-backend \
  --image gcr.io/reformatit/reformatit-backend \
  --platform managed \
  --region us-central1 \
  --allow-unauthenticated
